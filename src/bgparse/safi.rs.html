#[derive(PartialEq, Eq, Clone, Copy)]
pub struct Safi(u8);

impl From<u8> for Safi {
    fn from(other: u8) -> Safi {
        Safi(other)
    }
}

impl fmt::Debug for Safi {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        match self.0 {
            1 => write!(f, "unicast"),
            2 => write!(f, "multicast"),
            n => write!(f, "unknown({})", n),
        }
    }
}
